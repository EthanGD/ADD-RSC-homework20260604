### 2. 查看 `DiffTransformerLayer` 的定义

```python
class DiffTransformerLayer(nn.Module):
    def __init__(self, d_model, num_heads, depth):
        self.norm1 = RMSNorm(d_model)
        self.attn = MultiHeadDifferentialAttention(d_model, num_heads, depth)  # ← DDL的核心（差分注意力）
        self.norm2 = RMSNorm(d_model)
        self.ff = SwiGLU(d_model)
        self.classifier = nn.Linear(256, 4)
        self.aff = AFNO1D(hidden_size=d_model)  # ← 这就是 AFF（自适应频域滤波器）

    def forward(self, x):
        x1 = self.aff(x)  # ← 先经过 AFF 滤波
        x2 = (x1[:, 0] + x1[:, 1]) / 2
        x2 = self.classifier(x2)
        y = self.attn(self.norm1(x1)) + x1  # ← 再经过 DDL（差分注意力）
        z = self.ff(self.norm2(y)) + y
        return z, x2
```
